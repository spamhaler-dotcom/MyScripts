local function fmt(n)
	local s=tostring(math.floor(math.max(0,n)))
	return s:reverse():gsub("(%d%d%d)","%1,"):reverse():gsub("^,","")
end
local function parseN(str)
	if not str then return nil end
	local s=tostring(str):gsub(",",""):gsub("%s","")
	local n,suf=s:match("^(%d+%.?%d*)([KkMmBb]?)$")
	if n then
		n=tonumber(n)
		if suf=="K"or suf=="k" then n=n*1e3
		elseif suf=="M"or suf=="m" then n=n*1e6
		elseif suf=="B"or suf=="b" then n=n*1e9 end
		return math.floor(n)
	end
	return tonumber(s)
end

local function parseRaisedNum(t)
	local clean = t:gsub("[^%d%.KkMmBb]", "")
	local num, suffix = clean:match("^([%d%.]+)([KkMmBb]?)$")
	if not num then return 0 end
	num = tonumber(num) or 0
	if suffix=="K" or suffix=="k" then num=num*1e3
	elseif suffix=="M" or suffix=="m" then num=num*1e6
	elseif suffix=="B" or suffix=="b" then num=num*1e9 end
	return math.floor(num)
end

local function formatNum(n)
	if n>=1e9 then return string.format("%.3gB", n/1e9)
	elseif n>=1e6 then return string.format("%.3gM", n/1e6)
	elseif n>=1e3 then return string.format("%.3gK", n/1e3)
	else return tostring(math.floor(n)) end
end

local function getInfo(itemFrame,forcePrice,forceAsset)
	local info={name="Donation",price=forcePrice or 0,username="Player",assetId=forceAsset,imageId=nil}
	if not itemFrame then return info end
	pcall(function() info.assetId=itemFrame:GetAttribute("AssetId") or info.assetId; local n=itemFrame:GetAttribute("AssetName"); if n and n~="" then info.name=n end; local cn=itemFrame:GetAttribute("CreatorName"); if cn and cn~="" then info.username=cn end end)

	-- read attribute price before GetProductInfo overwrites it
	local attrPrice = 0
	pcall(function()
		local cur = itemFrame
		for _ = 1, 8 do
			local v = cur:GetAttribute("AssetPrice")
			if v and v > 0 then attrPrice = v; break end
			if not cur.Parent then break end
			cur = cur.Parent
		end
	end)
	info.price = attrPrice

	pcall(function() local cid=itemFrame:GetAttribute("CreatorId"); if cid and cid>0 then local ok,name=pcall(function() return Players:GetNameFromUserIdAsync(cid) end); if ok and name then info.username=name end end end)
	if info.username=="Player" then pcall(function() local cur=itemFrame; for _=1,12 do cur=cur.Parent; if not cur then break end; for _,d in pairs(cur:GetDescendants()) do if d:IsA("TextLabel") and d.Text and d.Visible then local m=d.Text:match("@([%w_]+)"); if m and #m>=3 then info.username=m; return end end end end end) end
	if info.assetId then
		local ok, prod = pcall(function() return MarketplaceService:GetProductInfo(info.assetId, Enum.InfoType.GamePass) end)
		print("GetProductInfo ok:", ok, "PriceInRobux:", ok and prod and prod.PriceInRobux)
		if ok and prod then
			if prod.Name and prod.Name~="" then info.name=prod.Name end
			if prod.IconImageAssetId and prod.IconImageAssetId>0 then info.imageId=prod.IconImageAssetId end
			if prod.Creator and prod.Creator.Name then info.username=prod.Creator.Name end
			if attrPrice > 0 then
				info.price = attrPrice  -- use it directly, no division
			elseif prod.PriceInRobux and prod.PriceInRobux > 0 then
				info.price = prod.PriceInRobux
			end
		else
			-- GetProductInfo failed, reverse the 40% cut
			if attrPrice > 0 then
				info.price = attrPrice
			end
		end
	end
	print("Final price:", info.price, "assetId:", info.assetId)
	return info
end

local currentPurchaseData = nil

purchaseGui.Enabled = false

local function showPurchaseUI(data)
	if isPurchaseOpen then 
		hidePurchaseUI()
		task.wait(0.1)
	end
	isPurchaseOpen = true
	currentPurchaseData = data

	closeRealPrompts()

	-- Clean username (no 's stand, no spaces)
	local cleanUsername = data.username or "Playername"
	cleanUsername = cleanUsername:gsub("'s [Ss]tand", ""):gsub("'s", ""):gsub("%s+$", "")

	-- Display name fallback chain
	local displayName = data.name
	if not displayName or displayName == "Donation" or displayName == "" then
		displayName = cleanUsername
	end

	-- Update text fields
	playerNameLbl.Text = displayName
	userHandleLbl.Text = "@" .. cleanUsername
	amountText.Text = fmt(data.price or 0)

	-- Update avatar (deferred)
	task.spawn(function()
		local targetUserId = data.userId
		local targetUsername = cleanUsername

		if not targetUserId and targetUsername and targetUsername ~= "Player" and targetUsername ~= "Playername" then
			local ok, uid = pcall(function() return Players:GetUserIdFromNameAsync(targetUsername) end)
			if ok and uid then targetUserId = uid end
		end

		if targetUserId then
			local ok, thumb = pcall(function()
				return Players:GetUserThumbnailAsync(targetUserId, Enum.ThumbnailType.HeadShot, Enum.ThumbnailSize.Size150x150)
			end)
			if ok and thumb and thumb ~= "" then
				avatarImg.Image = thumb
				defaultAvatarIcon.Visible = false
				return
			end
		end

		-- Fallbacks
		if data.imageId and data.imageId > 0 then
			avatarImg.Image = "rbxassetid://" .. tostring(data.imageId)
			defaultAvatarIcon.Visible = false
		elseif data.assetId then
			avatarImg.Image = "rbxthumb://type=GamePass&id=" .. tostring(data.assetId) .. "&w=150&h=150"
			defaultAvatarIcon.Visible = false
		else
			avatarImg.Image = ""
			defaultAvatarIcon.Visible = true
		end
	end)

	-- Update fee text
	local fee = math.floor((data.price or 0) * 0.1)
	local afterFee = (data.price or 0) - fee
	feeText.Text = "The recipient will get " .. fmt(afterFee) .. " Robux after fees. In-game transfers include a 10% creator fee."

	-- Animate in
	purchaseGui.Enabled = true
	overlay.BackgroundTransparency = 1
	card.Position = UDim2.new(0.5, 0, 0.55, 0)
	card.BackgroundTransparency = 0.1

	TweenService:Create(overlay, TweenInfo.new(0.2), {BackgroundTransparency = 0.4}):Play()
	TweenService:Create(card, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
		Position = UDim2.new(0.5, 0, 0.5, 0),
		BackgroundTransparency = 0
	}):Play()
end
